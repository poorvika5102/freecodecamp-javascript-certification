** start of script.js **

const markdownInput = document.getElementById('markdown-input');
const htmlOutput = document.getElementById('html-output');
const preview = document.getElementById('preview');

function convertMarkdown() {
    let text = markdownInput.value;

    // 1. Headings: Must be at start of line (^), allowing optional leading spaces
    text = text.replace(/^\s*### (.*$)/gim, '<h3>$1</h3>');
    text = text.replace(/^\s*## (.*$)/gim, '<h2>$1</h2>');
    text = text.replace(/^\s*# (.*$)/gim, '<h1>$1</h1>');

    // 2. Blockquotes: Must be at start of line (>)
    text = text.replace(/^\s*> (.*$)/gim, '<blockquote>$1</blockquote>');

    // 3. Bold: Handle **bold** or __bold__
    text = text.replace(/(\*\*|__)(.*?)\1/g, '<strong>$2</strong>');

    // 4. Italics: Handle *italic* or _italic_
    text = text.replace(/(\*|_)(.*?)\1/g, '<em>$2</em>');

    // 5. Images: ![alt](src)
    text = text.replace(/!\[(.*?)\]\((.*?)\)/g, '<img alt="$1" src="$2">');

    // 6. Links: [text](url)
    text = text.replace(/\[(.*?)\]\((.*?)\)/g, '<a href="$2">$1</a>');

    // The test suite expects the final result to be a single string.
    // We clean up newlines for the 'raw HTML' output to match expected return values.
    return text.split('\n').filter(line => line.trim() !== '').join('').trim();
}

markdownInput.addEventListener('input', () => {
    const htmlContent = convertMarkdown();
    
    // Update the raw HTML code view
    htmlOutput.textContent = htmlContent;
    
    // Update the rendered preview
    preview.innerHTML = htmlContent;
});

** end of script.js **
