** start of script.js **

const citySelect = document.getElementById('city-select');
const getWeatherBtn = document.getElementById('get-weather-btn');
const weatherDisplay = document.getElementById('weather-display');

// Helper function to handle undefined values
const formatValue = (value, suffix = '') => {
    return (value === undefined || value === null) ? 'N/A' : `${value}${suffix}`;
};

// 1. Asynchronous function to fetch data
async function getWeather(city) {
    try {
        const response = await fetch(`https://weather-proxy.freecodecamp.rocks/api/city/${city}`);
        const data = await response.json();
        return data;
    } catch (error) {
        console.log(error); // Requirement: Log error to console
    }
}

// 2. Asynchronous function to display data
async function showWeather(city) {
    const data = await getWeather(city);

    if (!data) {
        alert("Something went wrong, please try again later.");
        return;
    }

    // Unhide display
    weatherDisplay.classList.remove('hidden');

    // Populate elements using IDs defined in the user stories
    document.getElementById('location').innerText = formatValue(data.name);
    document.getElementById('weather-icon').src = data.weather?.[0]?.icon || '';
    document.getElementById('weather-main').innerText = formatValue(data.weather?.[0]?.main);
    document.getElementById('main-temperature').innerText = formatValue(data.main?.temp, '°C');
    document.getElementById('feels-like').innerText = formatValue(data.main?.feels_like, '°C');
    document.getElementById('humidity').innerText = formatValue(data.main?.humidity, '%');
    document.getElementById('wind').innerText = formatValue(data.wind?.speed, ' m/s');
    document.getElementById('wind-gust').innerText = formatValue(data.wind?.gust, ' m/s');
}

// Event listener for button
getWeatherBtn.addEventListener('click', () => {
    const selectedCity = citySelect.value;
    if (selectedCity !== "") {
        showWeather(selectedCity);
    }
});
