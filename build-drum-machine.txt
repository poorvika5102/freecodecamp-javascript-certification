** start of script.js **

const display = document.getElementById('display');

// Main function to play audio and update display
function handlePlay(element) {
    const audio = element.querySelector('.clip');
    if (audio) {
        audio.currentTime = 0;
        audio.play();
        // Set display to the ID of the button (the descriptive name)
        display.innerText = element.id.replace(/-/g, ' ');
    }
}

// 1. Click Event Listeners
document.querySelectorAll('.drum-pad').forEach(pad => {
    pad.addEventListener('click', () => {
        handlePlay(pad);
    });
});

// 2. Keyboard Event Listener
document.addEventListener('keydown', (event) => {
    const key = event.key.toUpperCase();
    const audio = document.getElementById(key);
    
    if (audio && audio.classList.contains('clip')) {
        const parent = audio.parentElement;
        handlePlay(parent);
        
        // Add visual feedback for keyboard press
        parent.style.backgroundColor = '#f8961e';
        setTimeout(() => {
            parent.style.backgroundColor = '';
        }, 100);
    }
});

** end of script.js **
